Основные команды и операторы

1. Оператор вывода данных

Для вывода данных на экран используется оператор Write или Writeln.

  • Write('Привет, мир!'); - вывод текста без перехода на новую строку.
  • Writeln('Привет, мир!'); - вывод текста с переходом на новую строку.

2. Оператор ввода данных

Для ввода данных используется оператор Read или Readln.

  • Read(a); - считывает значение в переменную a.
  • Readln(b); - считывает значение в переменную b и переходит на новую строку.

3. Условный оператор

Условный оператор позволяет выполнять разные блоки кода в зависимости от условия.

if x > 0 then
    Writeln('Число положительное')
else
    Writeln('Число отрицательное');

4. Циклы

Циклы позволяют повторять выполнение блока кода несколько раз.

  • Цикл for:
    • for i := 1 to 10 do
    Writeln('Итерация ', i);
  • Цикл while:
    • while x > 0 do
begin
    Writeln(x);
    x := x - 1;
end;
  • Цикл repeat:
    • repeat
    Writeln('Введите положительное число:');
    Readln(x);
until x > 0;

5. Массивы

Массивы используются для хранения набора значений одного типа.

var
    arr: array[1..5] of Integer;
begin
    arr[1] := 10;
    arr[2] := 20;
    arr[3] := 30;
    arr[4] := 40;
    arr[5] := 50;

    for i := 1 to 5 do
        Writeln('arr[', i, '] = ', arr[i]);
end.

6. Процедуры и функции

Процедуры и функции позволяют организовать код в виде повторно используемых блоков.

procedure SayHello(name: String);
begin
    Writeln('Привет, ', name, '!');
end;

function Sum(a, b: Integer): Integer;
begin
    Sum := a + b;
end;

begin
    SayHello('Вася');
    Writeln('Сумма: ', Sum(5, 7));
end.

7. Работа с файлами

Для работы с файлами используются специальные команды.

var
    f: Text;
    s: String;
begin
    Assign(f, 'file.txt');
    Rewrite(f);
    Writeln(f, 'Привет из файла!');
    Close(f);

    Reset(f);
    while not Eof(f) do
    begin
        Readln(f, s);
        Writeln(s);
    end;
    Close(f);
end.

8. Графика

PascalABC.net поддерживает работу с графикой через модуль GraphABC.

uses GraphABC;

begin
    SetWindowCaption('Графика в PascalABC');
    SetWindowSize(800, 600);

    ClearWindow(CornflowerBlue);
    DrawCircle(400, 300, 100);
    FloodFill(400, 300, Gold);
end.

9. Для изменений

PascalABC.net поддерживает работу с копированием текста PascalABC.net.

program kosmos;
uses GraphABC;

begin
  SetWindowSize(600, 450); // Размер окна, подобран визуально

  // Цвет контура и толщина
  SetPenColor(clBrown); 
  SetPenWidth(2);

  // *** Рисование ракеты ***
  // Корпус ракеты (трапеция)
  Line(100, 80, 140, 80); 
  Line(140, 80, 160, 220);
  Line(160, 220, 80, 220);
  Line(80, 220, 100, 80);

  // "Крылья" ракеты
  Line(80,220, 40,320);
  Line(40,320, 80,280);

  Line(160,220, 200,320);
  Line(200,320, 160,280);

  //Двигатели
  Line(100, 320, 110, 400);
  Line(110, 400, 120, 320);

  Line(130, 320, 140, 400);
  Line(140, 400, 150, 320);

  //Верхняя часть ракеты
  Line(100,80, 120,40);
  Line(120,40, 140,80);


  //Илюминаторы
  Circle(120, 120, 10);
  Circle(120, 180, 10);

  // *** Рисование звезд ***
  // Звезда слева внизу
  Line(20, 280, 40, 300);
  Line(40, 300, 60, 280);
  Line(60, 280, 40, 260);
  Line(40, 260, 20, 280);

  Line(40, 260, 40, 320);
  Line(40, 320, 40, 260);


  // *** Рисование луны ***
  Arc(500, 80, 30, 120, 300); // Центр, радиус, начальный угол, конечный угол


  // *** Рисование планет (окружности) ***
  Circle(40, 40, 10);     // Верхняя левая
  Circle(120, 160, 10);   // В центре
  Circle(340, 40, 10);  //В верхнем правом углу
  Circle(340,360,10);   //Средний правый
  Circle(80,360,10);   //Нижний левый
  Circle(500, 280, 10); //Правый нижний
  Circle(500,400,10); //Правый нижний
  
end.
