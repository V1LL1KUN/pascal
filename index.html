Основные команды и операторы

1. Оператор вывода данных

Для вывода данных на экран используется оператор Write или Writeln.

  • Write('Привет, мир!'); - вывод текста без перехода на новую строку.
  • Writeln('Привет, мир!'); - вывод текста с переходом на новую строку.

2. Оператор ввода данных

Для ввода данных используется оператор Read или Readln.

  • Read(a); - считывает значение в переменную a.
  • Readln(b); - считывает значение в переменную b и переходит на новую строку.

3. Условный оператор

Условный оператор позволяет выполнять разные блоки кода в зависимости от условия.

if x > 0 then
    Writeln('Число положительное')
else
    Writeln('Число отрицательное');

4. Циклы

Циклы позволяют повторять выполнение блока кода несколько раз.

  • Цикл for:
    • for i := 1 to 10 do
    Writeln('Итерация ', i);
  • Цикл while:
    • while x > 0 do
begin
    Writeln(x);
    x := x - 1;
end;
  • Цикл repeat:
    • repeat
    Writeln('Введите положительное число:');
    Readln(x);
until x > 0;

5. Массивы

Массивы используются для хранения набора значений одного типа.

var
    arr: array[1..5] of Integer;
begin
    arr[1] := 10;
    arr[2] := 20;
    arr[3] := 30;
    arr[4] := 40;
    arr[5] := 50;

    for i := 1 to 5 do
        Writeln('arr[', i, '] = ', arr[i]);
end.

6. Процедуры и функции

Процедуры и функции позволяют организовать код в виде повторно используемых блоков.

procedure SayHello(name: String);
begin
    Writeln('Привет, ', name, '!');
end;

function Sum(a, b: Integer): Integer;
begin
    Sum := a + b;
end;

begin
    SayHello('Вася');
    Writeln('Сумма: ', Sum(5, 7));
end.

7. Работа с файлами

Для работы с файлами используются специальные команды.

var
    f: Text;
    s: String;
begin
    Assign(f, 'file.txt');
    Rewrite(f);
    Writeln(f, 'Привет из файла!');
    Close(f);

    Reset(f);
    while not Eof(f) do
    begin
        Readln(f, s);
        Writeln(s);
    end;
    Close(f);
end.

8. Графика

PascalABC.net поддерживает работу с графикой через модуль GraphABC.

uses GraphABC;

begin
    SetWindowCaption('Графика в PascalABC');
    SetWindowSize(800, 600);

    ClearWindow(CornflowerBlue);
    DrawCircle(400, 300, 100);
    FloodFill(400, 300, Gold);
end.

9. Для изменений

PascalABC.net поддерживает работу с копированием текста PascalABC.net.

program kosmos;
uses GraphABC;

const
  CELL_SIZE = 20; // Размер клетки сетки
  GRID_COLOR = clGray; // Цвет сетки
  LINE_COLOR = clBrown; // Цвет линий
  LINE_WIDTH = 2; // Толщина линий

// Функция для преобразования координат сетки в координаты окна
function gridToPixelX(gridX: integer): integer;
begin
  gridToPixelX := gridX * CELL_SIZE;
end;

function gridToPixelY(gridY: integer): integer;
begin
  gridToPixelY := gridY * CELL_SIZE;
end;

var
  i: integer;

begin
  SetWindowSize(540, 420);
  SetPenColor(LINE_COLOR);
  SetPenWidth(LINE_WIDTH);

  // Рисуем координатную сетку (опционально)
  SetPenColor(GRID_COLOR);
  SetPenWidth(1);
  for i := 0 to 540 div CELL_SIZE do // 540 div 20 = 27
  begin
    Line(i * CELL_SIZE, 0, i * CELL_SIZE, 420);
  end;
  for i := 0 to 420 div CELL_SIZE do // 420 div 20 = 21
  begin
    Line(0, i * CELL_SIZE, 540, i * CELL_SIZE);
  end;

  SetPenColor(LINE_COLOR);
  SetPenWidth(LINE_WIDTH);


  // * Рисование ракеты *
  // Корпус ракеты (трапеция)
  Line(gridToPixelX(8), gridToPixelY(8), gridToPixelX(14), gridToPixelY(8)); // 8*20 = 160, 8*20 = 160 | 14*20 = 280, 8*20 = 160
  Line(gridToPixelX(14), gridToPixelY(8), gridToPixelX(16), gridToPixelY(22));
  Line(gridToPixelX(16), gridToPixelY(22), gridToPixelX(8), gridToPixelY(22));
  Line(gridToPixelX(8), gridToPixelY(22), gridToPixelX(8), gridToPixelY(8));

  // "Крылья" ракеты
  Line(gridToPixelX(8), gridToPixelY(22), gridToPixelX(4), gridToPixelY(32));
  Line(gridToPixelX(4), gridToPixelY(32), gridToPixelX(8), gridToPixelY(28));

  Line(gridToPixelX(16), gridToPixelY(22), gridToPixelX(20), gridToPixelY(32));
  Line(gridToPixelX(20), gridToPixelY(32), gridToPixelX(16), gridToPixelY(28));

  // Двигатели
  Line(gridToPixelX(10), gridToPixelY(32), gridToPixelX(11), gridToPixelY(40));
  Line(gridToPixelX(11), gridToPixelY(40), gridToPixelX(12), gridToPixelY(32));

  Line(gridToPixelX(13), gridToPixelY(32), gridToPixelX(14), gridToPixelY(40));
  Line(gridToPixelX(14), gridToPixelY(40), gridToPixelX(15), gridToPixelY(32));

  // Верхняя часть ракеты
  Line(gridToPixelX(8), gridToPixelY(8), gridToPixelX(12), gridToPixelY(4));
  Line(gridToPixelX(12), gridToPixelY(4), gridToPixelX(14), gridToPixelY(8));

  // Иллюминаторы
  Circle(gridToPixelX(12), gridToPixelY(12), 10);
  Circle(gridToPixelX(12), gridToPixelY(17), 10);
  Circle(gridToPixelX(12), gridToPixelY(6), 10);


  // * Рисование звезд *
  // Звезда слева внизу
  Line(gridToPixelX(2), gridToPixelY(28), gridToPixelX(4), gridToPixelY(30));
  Line(gridToPixelX(4), gridToPixelY(30), gridToPixelX(6), gridToPixelY(28));
  Line(gridToPixelX(6), gridToPixelY(28), gridToPixelX(4), gridToPixelY(26));
  Line(gridToPixelX(4), gridToPixelY(26), gridToPixelX(2), gridToPixelY(28));
  Line(gridToPixelX(4), gridToPixelY(26), gridToPixelX(4), gridToPixelY(32));
  Line(gridToPixelX(4), gridToPixelY(32), gridToPixelX(4), gridToPixelY(26));

  // * Рисование луны *
  Arc(gridToPixelX(45), gridToPixelY(8), 30, 120, 300); // Центр, радиус, начальный угол, конечный угол

  // * Рисование планет (окружности) *
  Circle(gridToPixelX(2), gridToPixelY(2), 10);     // Верхняя левая
  Circle(gridToPixelX(6), gridToPixelY(16), 10);   // В центре
  Circle(gridToPixelX(17), gridToPixelY(2), 10);  //В верхнем правом углу
  Circle(gridToPixelX(17), gridToPixelY(36), 10);   //Средний правый
  Circle(gridToPixelX(4), gridToPixelY(36), 10);   //Нижний левый
  Circle(gridToPixelX(25), gridToPixelY(28), 10); //Правый нижний
  Circle(gridToPixelX(25), gridToPixelY(20), 10); //Правый нижний
end.
