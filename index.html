Основные команды и операторы

1. Оператор вывода данных

Для вывода данных на экран используется оператор Write или Writeln.

  • Write('Привет, мир!'); - вывод текста без перехода на новую строку.
  • Writeln('Привет, мир!'); - вывод текста с переходом на новую строку.

2. Оператор ввода данных

Для ввода данных используется оператор Read или Readln.

  • Read(a); - считывает значение в переменную a.
  • Readln(b); - считывает значение в переменную b и переходит на новую строку.

3. Условный оператор

Условный оператор позволяет выполнять разные блоки кода в зависимости от условия.

if x > 0 then
    Writeln('Число положительное')
else
    Writeln('Число отрицательное');

4. Циклы

Циклы позволяют повторять выполнение блока кода несколько раз.

  • Цикл for:
    • for i := 1 to 10 do
    Writeln('Итерация ', i);
  • Цикл while:
    • while x > 0 do
begin
    Writeln(x);
    x := x - 1;
end;
  • Цикл repeat:
    • repeat
    Writeln('Введите положительное число:');
    Readln(x);
until x > 0;

5. Массивы

Массивы используются для хранения набора значений одного типа.

var
    arr: array[1..5] of Integer;
begin
    arr[1] := 10;
    arr[2] := 20;
    arr[3] := 30;
    arr[4] := 40;
    arr[5] := 50;

    for i := 1 to 5 do
        Writeln('arr[', i, '] = ', arr[i]);
end.

6. Процедуры и функции

Процедуры и функции позволяют организовать код в виде повторно используемых блоков.

procedure SayHello(name: String);
begin
    Writeln('Привет, ', name, '!');
end;

function Sum(a, b: Integer): Integer;
begin
    Sum := a + b;
end;

begin
    SayHello('Вася');
    Writeln('Сумма: ', Sum(5, 7));
end.

7. Работа с файлами

Для работы с файлами используются специальные команды.

var
    f: Text;
    s: String;
begin
    Assign(f, 'file.txt');
    Rewrite(f);
    Writeln(f, 'Привет из файла!');
    Close(f);

    Reset(f);
    while not Eof(f) do
    begin
        Readln(f, s);
        Writeln(s);
    end;
    Close(f);
end.

8. Графика

PascalABC.net поддерживает работу с графикой через модуль GraphABC.

uses GraphABC;

begin
    SetWindowCaption('Графика в PascalABC');
    SetWindowSize(800, 600);

    ClearWindow(CornflowerBlue);
    DrawCircle(400, 300, 100);
    FloodFill(400, 300, Gold);
end.

9. Для изменений

PascalABC.net поддерживает работу с копированием текста PascalABC.net.

program oreshnick;
uses graphabc;

begin
  setwindowsize(540, 540);
  setpencolor(clblack);
  setpenwidth(3); // Thinner lines look more like the image

  // Rocket body
  moveto(179, 59);
  lineto(235, 91);
  lineto(286, 245);
  lineto(225, 279);
  lineto(159, 120);
  lineto(179, 59);

  // Rocket wing 1
  moveto(260, 170);
  lineto(335, 190);
  lineto(360, 230);
  lineto(279, 250);

  // Rocket wing 2
  moveto(187, 195);
  lineto(160, 240);

  //Rocket flame
  lineto(155, 297);
  lineto(225, 277);
  lineto(222, 343);
  lineto(244, 315);
  lineto(265, 375);
  lineto(275, 315);
  lineto(315, 367);
  lineto(299, 298);
  lineto(335, 315);
  lineto(286, 245);

  // Windows
  circle(200, 125, 15); // Smaller circles for the windows
  circle(220, 170, 15);
  circle(240, 225, 15);

  // Filling
  floodfill(236, 110, clyellow);  // Filling around the rocket Yellow like in image

  floodfill(290, 210, clyellow);  //Filling around the rocket Yellow like in image
  floodfill(170, 265, clyellow);  //Filling around the rocket Yellow like in image
  floodfill(200, 125, clyellow);  //Windows Yellow
  floodfill(220, 170, clyellow);  //Windows Yellow
  floodfill(240, 225, clyellow);  //Windows Yellow

end.
